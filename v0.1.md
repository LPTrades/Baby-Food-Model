"""

Swing + First FVG Pine Script Exporter

=======================================

Outputs a Pine Script that combines:

  

1. Most recent 5m swing high (red dashed)

2. Most recent 5m swing low (green dashed)

3. Broken swing level (dotted line)

4. First 1m FVG after the break (box) — stays until next break

  

Fix: recentHigh/recentLow cleared after break so the break

condition only fires once per swing level.

"""

  

from pathlib import Path

  

# =============================================================================

# CONFIG

# =============================================================================

CONFIG = {

    "session":     "0930-1200",

    "pine_output": "swing_fvg.pine",

}

  

# =============================================================================

# EXPORT

# =============================================================================

def export_pine(session: str, output_path: str):

    pine = f"""//@version=5

indicator("Swing → First FVG Only", overlay=true, max_boxes_count=500, max_lines_count=500)

  

// ==============================

// SESSION

// ==============================

in_session   = not na(time("1", "{session}", "America/New_York"))

session_open = in_session and not in_session[1]

is_1m        = timeframe.period == "1"

  

// ==============================

// 5M SWINGS

// ==============================

[h5, l5]     = request.security(syminfo.tickerid, "5", [high,    low],    lookahead=barmerge.lookahead_off)

[h5_1, l5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1]], lookahead=barmerge.lookahead_off)

[h5_2, l5_2] = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

  

is_swing_high = h5_1 > h5_2 and h5_1 > h5

is_swing_low  = l5_1 < l5_2 and l5_1 < l5

  

// ==============================

// STATE

// ==============================

var float recentHigh  = na

var float recentLow   = na

var float brokenLevel = na

var line  highLine    = na

var line  lowLine     = na

var line  brokenLine  = na

var box   fvgBox      = na

var int   mode        = 0

var bool  fvgFound    = false

  

// ==============================

// SESSION RESET

// ==============================

if session_open

    recentHigh  := na

    recentLow   := na

    brokenLevel := na

    mode        := 0

    fvgFound    := false

    if not na(highLine)

        line.delete(highLine)

        highLine := na

    if not na(lowLine)

        line.delete(lowLine)

        lowLine := na

    if not na(brokenLine)

        line.delete(brokenLine)

        brokenLine := na

    if not na(fvgBox)

        box.delete(fvgBox)

        fvgBox := na

  

// ==============================

// UPDATE SWING LINES

// ==============================

if in_session

  

    if is_swing_high

        recentHigh := h5_1

        if not na(highLine)

            line.delete(highLine)

        highLine := line.new(

             x1    = bar_index,

             y1    = recentHigh,

             x2    = bar_index + 300,

             y2    = recentHigh,

             color = color.red,

             width = 2,

             style = line.style_dashed,

             xloc  = xloc.bar_index

             )

  

    if is_swing_low

        recentLow := l5_1

        if not na(lowLine)

            line.delete(lowLine)

        lowLine := line.new(

             x1    = bar_index,

             y1    = recentLow,

             x2    = bar_index + 300,

             y2    = recentLow,

             color = color.green,

             width = 2,

             style = line.style_dashed,

             xloc  = xloc.bar_index

             )

  

// ==============================

// BREAK DETECTION

// Clear the level after break so this only fires once

// ==============================

if in_session

  

    if not na(recentHigh) and high > recentHigh

        mode        := -1

        fvgFound    := false

        brokenLevel := recentHigh

        recentHigh  := na        // clear so break doesn't keep firing

        if not na(fvgBox)

            box.delete(fvgBox)

            fvgBox := na

        if not na(brokenLine)

            line.delete(brokenLine)

        brokenLine := line.new(

             x1    = bar_index,

             y1    = brokenLevel,

             x2    = bar_index + 300,

             y2    = brokenLevel,

             color = color.red,

             width = 2,

             style = line.style_dotted,

             xloc  = xloc.bar_index

             )

  

    if not na(recentLow) and low < recentLow

        mode        := 1

        fvgFound    := false

        brokenLevel := recentLow

        recentLow   := na        // clear so break doesn't keep firing

        if not na(fvgBox)

            box.delete(fvgBox)

            fvgBox := na

        if not na(brokenLine)

            line.delete(brokenLine)

        brokenLine := line.new(

             x1    = bar_index,

             y1    = brokenLevel,

             x2    = bar_index + 300,

             y2    = brokenLevel,

             color = color.green,

             width = 2,

             style = line.style_dotted,

             xloc  = xloc.bar_index

             )

  

// ==============================

// FIRST FVG AFTER BREAK — 1m only

// Box stays until next break

// ==============================

bull_fvg = low > high[2]

bear_fvg = high < low[2]

  

if is_1m and in_session and mode != 0 and not fvgFound

  

    if mode == 1 and bull_fvg

        if not na(fvgBox)

            box.delete(fvgBox)

        fvgBox   := box.new(

             bar_index[2], low, bar_index, high[2],

             bgcolor      = color.new(color.green, 70),

             border_color = color.green

             )

        fvgFound := true

  

    if mode == -1 and bear_fvg

        if not na(fvgBox)

            box.delete(fvgBox)

        fvgBox   := box.new(

             bar_index[2], low[2], bar_index, high,

             bgcolor      = color.new(color.red, 70),

             border_color = color.red

             )

        fvgFound := true

"""

  

    with open(output_path, "w") as f:

        f.write(pine)

  

    print(f"[EXPORT] Pine Script saved to: {output_path}")

    print(f"         Paste into TradingView Pine Editor.")

    print(f"")

    print(f"         Red dashed   = most recent 5m swing high")

    print(f"         Green dashed = most recent 5m swing low")

    print(f"         Red dotted   = broken swing high level")

    print(f"         Green dotted = broken swing low level")

    print(f"         Green box    = first bullish FVG after swing low broken")

    print(f"         Red box      = first bearish FVG after swing high broken")

    print(f"         FVG box stays until next swing is broken.")

  

# =============================================================================

# MAIN

# =============================================================================

if __name__ == "__main__":

    session = input(f"Enter session window (default: {CONFIG['session']}): ").strip()

    if not session:

        session = CONFIG["session"]

  

    export_pine(session, CONFIG["pine_output"])
