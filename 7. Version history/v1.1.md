"""
5
=============================================================================

NQ FUTURES BACKTESTER

=============================================================================

Matches the Pine Script logic exactly:

  

1. Detect 5m swing highs and lows (immediate neighbours, high/low based)

2. Detect break of most recent swing

3. Track structural stop (highest high / lowest low from break to FVG)

4. Find first 1m FVG after break

5. Enter when price touches FVG entry level on a bar AFTER FVG forms

6. Stop = structural extreme of displacement leg

7. TP = 2R from entry

8. Close on TP or SL hit — exits are NOT session restricted

  

Entry sessions (NY time):

  03:00 - 04:00

  09:30 - 11:00

  14:00 - 15:00

  

Exits: hold until TP or SL regardless of time

  

Outputs:

  - Performance stats

  - 20 random examples from 2026

=============================================================================

"""

  

import pandas as pd

import numpy as np

import random

from pathlib import Path

  

DATA_PATH = r"C:\tradingProject\Data"

  

# =============================================================================

# CONFIG

# =============================================================================

CONFIG = {

    "account_size": 100000.0,

    "risk_pct":     0.01,       # 1% risk per trade

    "tp_r":         2.0,        # 2R take profit

}

  

# Entry session windows (NY time, 24hr format)

SESSIONS = [

    ("03:00", "04:00"),

    ("09:30", "11:00"),

    ("14:00", "15:00"),

]

  

# =============================================================================

# SESSION CHECK

# =============================================================================

def in_session(ts: pd.Timestamp) -> bool:

    t = ts.time()

    for start_str, end_str in SESSIONS:

        start = pd.Timestamp(start_str).time()

        end   = pd.Timestamp(end_str).time()

        if start <= t < end:

            return True

    return False

  

# =============================================================================

# LOAD DATA

# =============================================================================

def load_csv(filename: str) -> pd.DataFrame | None:

    path = Path(DATA_PATH) / filename

    if not path.exists():

        print(f"[ERROR] File not found: {path}")

        return None

  

    df = pd.read_csv(path)

    df.columns = df.columns.str.strip().str.lower()

  

    for alias in ["date", "timestamp", "time"]:

        if alias in df.columns and "datetime" not in df.columns:

            df.rename(columns={alias: "datetime"}, inplace=True)

  

    df["datetime"] = pd.to_datetime(df["datetime"])

    df.set_index("datetime", inplace=True)

    df.sort_index(inplace=True)

    df.dropna(subset=["open", "high", "low", "close"], inplace=True)

  

    print(f"[LOADED] {filename} | Rows: {len(df)} | {df.index[0]} to {df.index[-1]}")

    return df

  

# =============================================================================

# SWING DETECTION (5m)

# Swing high: high[i] > high[i-1] AND high[i] > high[i+1]

# Swing low:  low[i]  < low[i-1]  AND low[i]  < low[i+1]

# =============================================================================

def detect_swings_5m(df: pd.DataFrame) -> pd.DataFrame:

    df = df.copy()

    df["is_swing_high"]    = False

    df["is_swing_low"]     = False

    df["swing_high_price"] = np.nan

    df["swing_low_price"]  = np.nan

  

    highs = df["high"].values

    lows  = df["low"].values

    n = len(df)

  

    for i in range(1, n - 1):

        if highs[i] > highs[i - 1] and highs[i] > highs[i + 1]:

            df.iloc[i, df.columns.get_loc("is_swing_high")]    = True

            df.iloc[i, df.columns.get_loc("swing_high_price")] = highs[i]

  

        if lows[i] < lows[i - 1] and lows[i] < lows[i + 1]:

            df.iloc[i, df.columns.get_loc("is_swing_low")]    = True

            df.iloc[i, df.columns.get_loc("swing_low_price")] = lows[i]

  

    return df

  

# =============================================================================

# FVG DETECTION (1m)

# Bullish FVG: low[i+2] > high[i]

#   entry = low[i+2] = top of green box (price drops into gap)

# Bearish FVG: high[i+2] < low[i]

#   entry = high[i+2] = bottom of red box (price rises into gap)

# =============================================================================

def detect_fvg_1m(df: pd.DataFrame) -> pd.DataFrame:

    df = df.copy()

    n = len(df)

  

    bull     = np.zeros(n, dtype=bool)

    bear     = np.zeros(n, dtype=bool)

    bull_top = np.full(n, np.nan)

    bull_bot = np.full(n, np.nan)

    bear_top = np.full(n, np.nan)

    bear_bot = np.full(n, np.nan)

    c1_time  = [None] * n

    c2_time  = [None] * n

    c3_time  = [None] * n

  

    highs = df["high"].values

    lows  = df["low"].values

    times = df.index

  

    for i in range(n - 2):

        # Bullish FVG

        if lows[i + 2] > highs[i]:

            bull[i]     = True

            bull_top[i] = lows[i + 2]   # top of gap = entry

            bull_bot[i] = highs[i]      # bottom of gap

            c1_time[i]  = times[i]

            c2_time[i]  = times[i + 1]

            c3_time[i]  = times[i + 2]

  

        # Bearish FVG

        if highs[i + 2] < lows[i]:

            bear[i]     = True

            bear_top[i] = lows[i]       # top of gap

            bear_bot[i] = highs[i + 2] # bottom of gap = entry

            c1_time[i]  = times[i]

            c2_time[i]  = times[i + 1]

            c3_time[i]  = times[i + 2]

  

    df["fvg_bull"]     = bull

    df["fvg_bear"]     = bear

    df["fvg_bull_top"] = bull_top

    df["fvg_bull_bot"] = bull_bot

    df["fvg_bear_top"] = bear_top

    df["fvg_bear_bot"] = bear_bot

    df["fvg_c1_time"]  = c1_time

    df["fvg_c2_time"]  = c2_time

    df["fvg_c3_time"]  = c3_time

  

    return df

  

# =============================================================================

# MAIN BACKTEST

# =============================================================================

def run_backtest(df_1m: pd.DataFrame, df_5m: pd.DataFrame, cfg: dict) -> dict:

  

    df_5m = detect_swings_5m(df_5m)

    df_1m = detect_fvg_1m(df_1m)

  

    trades       = []

    equity       = cfg["account_size"]

    equity_curve = [equity]

  

    # Setup state

    recent_high  = None

    recent_low   = None

    mode         = 0

    fvg_found    = False

    entry_level  = None

    fvg_bar_time = None

    struct_stop  = None

    fvg_data     = None

  

    # Trade state

    trade_open       = False

    trade_mode       = 0

    trade_entry      = None

    trade_sl         = None

    trade_tp         = None

    trade_entry_time = None

    trade_fvg_data   = None

  

    for i, (ts, row_1m) in enumerate(df_1m.iterrows()):

        h = row_1m["high"]

        l = row_1m["low"]

  

        # ================================================================

        # EXITS — no session restriction, hold until TP or SL

        # ================================================================

        if trade_open:

            hit_tp = (trade_mode == 1  and h >= trade_tp) or \

                     (trade_mode == -1 and l <= trade_tp)

            hit_sl = (trade_mode == 1  and l <= trade_sl) or \

                     (trade_mode == -1 and h >= trade_sl)

  

            if hit_tp or hit_sl:

                outcome  = "win" if hit_tp else "loss"

                pnl_r    = cfg["tp_r"] if hit_tp else -1.0

                risk_amt = equity * cfg["risk_pct"]

                pnl_usd  = risk_amt * pnl_r

                equity  += pnl_usd

  

                trades.append({

                    "entry_time":  trade_entry_time,

                    "exit_time":   ts,

                    "direction":   "long" if trade_mode == 1 else "short",

                    "entry":       trade_entry,

                    "sl":          trade_sl,

                    "tp":          trade_tp,

                    "outcome":     outcome,

                    "r":           round(pnl_r, 2),

                    "pnl_usd":     round(pnl_usd, 2),

                    "equity":      round(equity, 2),

                    "fvg_c1":      trade_fvg_data["c1"] if trade_fvg_data else None,

                    "fvg_c2":      trade_fvg_data["c2"] if trade_fvg_data else None,

                    "fvg_c3":      trade_fvg_data["c3"] if trade_fvg_data else None,

                })

                equity_curve.append(equity)

  

                # Reset all state after trade closes

                trade_open   = False

                trade_mode   = 0

                mode         = 0

                fvg_found    = False

                entry_level  = None

                struct_stop  = None

                fvg_bar_time = None

                recent_high  = None

                recent_low   = None

  

            continue  # skip setup logic while in a trade

  

        # ================================================================

        # SETUP LOGIC — session restricted

        # ================================================================

        active = in_session(ts)

  

        # ---- Update most recent 5m swings ----

        past_5m = df_5m[df_5m.index <= ts]

        if len(past_5m) >= 2:

            last_sh = past_5m[past_5m["is_swing_high"]]["swing_high_price"].dropna()

            last_sl = past_5m[past_5m["is_swing_low"]]["swing_low_price"].dropna()

            if len(last_sh) > 0:

                recent_high = float(last_sh.iloc[-1])

            if len(last_sl) > 0:

                recent_low = float(last_sl.iloc[-1])

  

        # ---- Break detection — session only ----

        if active:

            if recent_high is not None and h > recent_high and mode != -1:

                mode         = -1

                fvg_found    = False

                entry_level  = None

                struct_stop  = None

                fvg_bar_time = None

                recent_high  = None

  

            if recent_low is not None and l < recent_low and mode != 1:

                mode         = 1

                fvg_found    = False

                entry_level  = None

                struct_stop  = None

                fvg_bar_time = None

                recent_low   = None

  

        # ---- Track structural stop — session only ----

        if active and mode == -1 and not fvg_found:

            struct_stop = h if struct_stop is None else max(struct_stop, h)

  

        if active and mode == 1 and not fvg_found:

            struct_stop = l if struct_stop is None else min(struct_stop, l)

  

        # ---- FVG detection — session only ----

        if active and mode != 0 and not fvg_found and i >= 2:

            fvg_c1_row = df_1m.iloc[i - 2]

  

            if mode == 1 and fvg_c1_row["fvg_bull"]:

                fvg_found    = True

                entry_level  = float(fvg_c1_row["fvg_bull_top"])

                fvg_bar_time = ts

                fvg_data     = {

                    "c1": str(fvg_c1_row["fvg_c1_time"]),

                    "c2": str(fvg_c1_row["fvg_c2_time"]),

                    "c3": str(fvg_c1_row["fvg_c3_time"]),

                }

  

            if mode == -1 and fvg_c1_row["fvg_bear"]:

                fvg_found    = True

                entry_level  = float(fvg_c1_row["fvg_bear_bot"])

                fvg_bar_time = ts

                fvg_data     = {

                    "c1": str(fvg_c1_row["fvg_c1_time"]),

                    "c2": str(fvg_c1_row["fvg_c2_time"]),

                    "c3": str(fvg_c1_row["fvg_c3_time"]),

                }

  

        # ---- Entry trigger — session only ----

        if active and fvg_found and entry_level is not None and \

           fvg_bar_time is not None and ts > fvg_bar_time and \

           struct_stop is not None:

  

            if mode == 1 and l <= entry_level:

                risk = abs(entry_level - struct_stop)

                if risk > 0:

                    trade_open       = True

                    trade_mode       = 1

                    trade_entry      = entry_level

                    trade_sl         = struct_stop

                    trade_tp         = entry_level + cfg["tp_r"] * risk

                    trade_entry_time = ts

                    trade_fvg_data   = fvg_data

                    fvg_found        = False

  

            elif mode == -1 and h >= entry_level:

                risk = abs(entry_level - struct_stop)

                if risk > 0:

                    trade_open       = True

                    trade_mode       = -1

                    trade_entry      = entry_level

                    trade_sl         = struct_stop

                    trade_tp         = entry_level - cfg["tp_r"] * risk

                    trade_entry_time = ts

                    trade_fvg_data   = fvg_data

                    fvg_found        = False

  

    return {

        "trades":       trades,

        "equity_curve": equity_curve,

        "final_equity": equity,

    }

def compute_stats(results: dict, cfg: dict) -> dict:

    trades = results["trades"]

    if not trades:

        return {"total_trades": 0}

  

    df = pd.DataFrame(trades)

    wins   = df[df["outcome"] == "win"]

    losses = df[df["outcome"] == "loss"]

  

    total      = len(df)

    win_count  = len(wins)

    loss_count = len(losses)

    win_rate   = win_count / total if total > 0 else 0

    wl_ratio   = win_count / loss_count if loss_count > 0 else float("inf")

    avg_r      = df["r"].mean()

    total_pnl  = df["pnl_usd"].sum()

  

    # Max drawdown from equity high

    eq   = np.array(results["equity_curve"])

    peak = np.maximum.accumulate(eq)

    dd   = (eq - peak) / peak * 100

    max_dd = dd.min()

  

    # Streaks

    outcomes        = df["outcome"].tolist()

    max_win_streak  = 0

    max_loss_streak = 0

    cur_win         = 0

    cur_loss        = 0

    for o in outcomes:

        if o == "win":

            cur_win  += 1

            cur_loss  = 0

            max_win_streak = max(max_win_streak, cur_win)

        else:

            cur_loss += 1

            cur_win   = 0

            max_loss_streak = max(max_loss_streak, cur_loss)

  

    return {

        "total_trades":        total,

        "wins":                win_count,

        "losses":              loss_count,

        "win_rate_%":          round(win_rate * 100, 2),

        "w/l_ratio":           round(wl_ratio, 2),

        "avg_r":               round(avg_r, 3),

        "total_pnl_usd":       round(total_pnl, 2),

        "final_equity":        round(results["final_equity"], 2),

        "max_drawdown_%":      round(max_dd, 2),

        "longest_win_streak":  max_win_streak,

        "longest_loss_streak": max_loss_streak,

    }

def print_stats(stats: dict):

    print("\n" + "=" * 55)

    print("  BACKTEST RESULTS")

    print("=" * 55)

    for k, v in stats.items():

        print(f"  {k.replace('_', ' ').title():<28} {v}")

    print("=" * 55)

def print_examples(trades: list, n: int = 20):

    trades_2026 = [t for t in trades if str(t["entry_time"]).startswith("2026")]

  

    if not trades_2026:

        print("\n[INFO] No trades found in 2026.")

        return

  

    sample = random.sample(trades_2026, min(n, len(trades_2026)))

    sample.sort(key=lambda x: x["entry_time"])

  

    print(f"\n{'=' * 90}")

    print(f"  {len(sample)} RANDOM EXAMPLES FROM 2026")

    print(f"{'=' * 90}")

  

    for i, t in enumerate(sample, 1):

        direction = t["direction"].upper()

        outcome   = t["outcome"].upper()

        print(f"\n  [{i}] {direction} | {outcome} | R: {t['r']}")

        print(f"      Entry Time : {t['entry_time']}")

        print(f"      Exit Time  : {t['exit_time']}")

        print(f"      Entry Price: {t['entry']:.2f}")

        print(f"      Stop Loss  : {t['sl']:.2f}")

        print(f"      Take Profit: {t['tp']:.2f}")

        print(f"      Risk       : {abs(t['entry'] - t['sl']):.2f} pts")

        if t["fvg_c1"]:

            print(f"      FVG C1     : {t['fvg_c1']}")

            print(f"      FVG C2     : {t['fvg_c2']}")

            print(f"      FVG C3     : {t['fvg_c3']}")

        print(f"      PnL        : ${t['pnl_usd']:,.2f}")

  

    print(f"\n{'=' * 90}")
