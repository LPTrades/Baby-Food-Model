```
//@version=5

strategy("5m Break → 1m CISD + FVG Strategy", overlay=true, max_lines_count=200, max_labels_count=200, default_qty_type=strategy.fixed, default_qty_value=1)

  

// ============================================================

// SETTINGS

// ============================================================

session_input    = input.session("0930-1100", "Entry Session (ET)")

session_full     = input.session("0930-1200", "Full Session (ET)")

  

// ============================================================

// SESSION HELPERS

// ============================================================

in_entry_session = not na(time("1", session_input, "America/New_York"))

in_full_session  = not na(time("1", session_full,  "America/New_York"))

session_end_time = timestamp("America/New_York", year, month, dayofmonth, 12, 0, 0)

  

// ============================================================

// 5m DATA

// ============================================================

[h5,  l5 ]      = request.security(syminfo.tickerid, "5", [high,    low   ], lookahead=barmerge.lookahead_off)

[h5_1, l5_1, t5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1], time[1]], lookahead=barmerge.lookahead_off)

[h5_2, l5_2]    = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

  

is_swing_high_5m = h5_1 > h5_2 and h5_1 > h5

is_swing_low_5m  = l5_1 < l5_2 and l5_1 < l5

  

new_5m_bar = ta.change(t5_1) != 0

  

// New session detection (1m)

new_session_1m = in_full_session and not in_full_session[1]

  

// ============================================================

// 5m SWING STATE

// ============================================================

var float recent_sh       = na

var float recent_sl       = na

var int   recent_sh_time  = na

var int   recent_sl_time  = na

  

var line  recent_sh_line  = na

var line  recent_sl_line  = na

var line  broken_sh_line  = na

var line  broken_sl_line  = na

  

// ============================================================

// STRATEGY STATE

// ============================================================

// Bias: 0=none, 1=look for long (swing low broken), -1=look for short (swing high broken)

var int   bias            = 0

var bool  setup_used      = false  // one setup per swing break

  

// CISD tracking

var int   cisd_start_idx  = na   // bar_index of first bearish(long) / bullish(short) candle

var int   cisd_end_idx    = na   // bar_index of last candle in the run

var float cisd_range_high = na   // open of first candle (bullish cisd top)

var float cisd_range_low  = na   // open of first candle (bearish cisd bottom)

var float cisd_sl_level   = na   // lowest close (long) / highest close (short)

  

// FVG tracking

var float fvg_high        = na

var float fvg_low         = na

var bool  fvg_valid       = false

  

// Trade levels

var float entry_price     = na

var float sl_price        = na

var float tp_price        = na

var bool  in_trade        = false

  

// Lines for trade levels

var line  sl_line         = na

var line  tp_line         = na

var line  cisd_line       = na

var line  cisd_range_line = na

  

// ============================================================

// RESET ON NEW SESSION

// ============================================================

if new_session_1m

    recent_sh      := na

    recent_sl      := na

    recent_sh_time := na

    recent_sl_time := na

    line.delete(recent_sh_line)

    line.delete(recent_sl_line)

    line.delete(broken_sh_line)

    line.delete(broken_sl_line)

    recent_sh_line := na

    recent_sl_line := na

    broken_sh_line := na

    broken_sl_line := na

    bias           := 0

    setup_used     := false

    cisd_start_idx := na

    cisd_end_idx   := na

    cisd_range_high := na

    cisd_range_low  := na

    cisd_sl_level   := na

    fvg_high       := na

    fvg_low        := na

    fvg_valid      := false

  

// ============================================================

// 5m SWING CAPTURE

// ============================================================

if in_full_session and new_5m_bar and is_swing_high_5m

    recent_sh      := h5_1

    recent_sh_time := t5_1

    line.delete(recent_sh_line)

    recent_sh_line := line.new(recent_sh_time, recent_sh, session_end_time, recent_sh, xloc=xloc.bar_time, color=color.red, style=line.style_dashed, width=1)

  

if in_full_session and new_5m_bar and is_swing_low_5m

    recent_sl      := l5_1

    recent_sl_time := t5_1

    line.delete(recent_sl_line)

    recent_sl_line := line.new(recent_sl_time, recent_sl, session_end_time, recent_sl, xloc=xloc.bar_time, color=color.green, style=line.style_dashed, width=1)

  

// ============================================================

// 5m BREAK DETECTION → SET BIAS

// ============================================================

// Swing HIGH broken → bearish bias (look for short)

if in_full_session and not na(recent_sh) and high > recent_sh and not in_trade

    line.delete(broken_sh_line)

    broken_sh_line := line.new(recent_sh_time, recent_sh, session_end_time, recent_sh, xloc=xloc.bar_time, color=color.red, style=line.style_dotted, width=1)

    line.delete(recent_sh_line)

    recent_sh_line := na

    recent_sh      := na

    if not setup_used

        bias       := -1

        setup_used := false

        // Reset CISD/FVG for new bias

        cisd_start_idx  := na

        cisd_end_idx    := na

        cisd_range_high := na

        cisd_range_low  := na

        cisd_sl_level   := na

        fvg_high        := na

        fvg_low         := na

        fvg_valid       := false

  

// Swing LOW broken → bullish bias (look for long)

if in_full_session and not na(recent_sl) and low < recent_sl and not in_trade

    line.delete(broken_sl_line)

    broken_sl_line := line.new(recent_sl_time, recent_sl, session_end_time, recent_sl, xloc=xloc.bar_time, color=color.green, style=line.style_dotted, width=1)

    line.delete(recent_sl_line)

    recent_sl_line := na

    recent_sl      := na

    if not setup_used

        bias       := 1

        setup_used := false

        cisd_start_idx  := na

        cisd_end_idx    := na

        cisd_range_high := na

        cisd_range_low  := na

        cisd_sl_level   := na

        fvg_high        := na

        fvg_low         := na

        fvg_valid       := false

  

// ============================================================

// CISD DETECTION (1m, during entry session, after bias set)

// Only track while NOT in a trade and setup not used

// ============================================================

  

// Helpers

is_bearish_1m = close < open

is_bullish_1m = close > open

  

// ----- BULLISH CISD (for LONG bias) -----

// Track consecutive bearish candles

var int   bear_run_start  = na

var int   bear_run_count  = 0

var float bear_run_open   = na   // open of first bearish candle

var float bear_run_sl     = na   // lowest close in the run

  

if in_entry_session and bias == 1 and not in_trade and not setup_used

    if is_bearish_1m

        if na(bear_run_start) or not is_bearish_1m[1]

            // Start new bearish run

            bear_run_start := bar_index

            bear_run_open  := open

            bear_run_sl    := close

            bear_run_count := 1

        else

            // Extend run

            bear_run_count += 1

            bear_run_sl    := math.min(bear_run_sl, close)

            // If new lower low, update CISD range and rescan FVG

            if low < cisd_range_low or na(cisd_range_low)

                cisd_start_idx  := bear_run_start

                cisd_end_idx    := bar_index

                cisd_range_high := bear_run_open

                cisd_range_low  := low

                cisd_sl_level   := bear_run_sl

                // Rescan FVG over new range

                fvg_valid  := false

                fvg_high   := na

                fvg_low    := na

  

        // Update CISD range on first capture

        if bear_run_count >= 2

            if na(cisd_start_idx)

                cisd_start_idx  := bear_run_start

                cisd_end_idx    := bar_index

                cisd_range_high := bear_run_open

                cisd_range_low  := low

                cisd_sl_level   := bear_run_sl

  

    else

        // Non-bearish candle resets the tracking for next sequence

        // (but keeps stored CISD range if already captured)

        bear_run_start := na

        bear_run_count := 0

        bear_run_open  := na

        bear_run_sl    := na

  

// ----- BEARISH CISD (for SHORT bias) -----

var int   bull_run_start  = na

var int   bull_run_count  = 0

var float bull_run_open   = na

var float bull_run_sl     = na   // highest close in the run

  

if in_entry_session and bias == -1 and not in_trade and not setup_used

    if is_bullish_1m

        if na(bull_run_start) or not is_bullish_1m[1]

            bull_run_start := bar_index

            bull_run_open  := open

            bull_run_sl    := close

            bull_run_count := 1

        else

            bull_run_count += 1

            bull_run_sl    := math.max(bull_run_sl, close)

            if high > cisd_range_high or na(cisd_range_high)

                cisd_start_idx  := bull_run_start

                cisd_end_idx    := bar_index

                cisd_range_low  := bull_run_open

                cisd_range_high := high

                cisd_sl_level   := bull_run_sl

                fvg_valid  := false

                fvg_high   := na

                fvg_low    := na

  

        if bull_run_count >= 2

            if na(cisd_start_idx)

                cisd_start_idx  := bull_run_start

                cisd_end_idx    := bar_index

                cisd_range_low  := bull_run_open

                cisd_range_high := high

                cisd_sl_level   := bull_run_sl

  

    else

        bull_run_start := na

        bull_run_count := 0

        bull_run_open  := na

        bull_run_sl    := na

  

// ============================================================

// FVG SCAN — scan backward from cisd_end_idx each bar

// Only store the FIRST (earliest) valid FVG in the range

// Only candle 2 must belong to the active CISD range

// ============================================================

scan_bars = not na(cisd_start_idx) and not fvg_valid and not in_trade and not setup_used

  

if scan_bars and bar_index >= cisd_start_idx + 2

    // Only scan if not already found a valid FVG

    // We scan the most recent three-bar window where candle[1] is in CISD range

    // To find the FIRST FVG, we scan from cisd_start_idx+1 forward

    // Approximation: check current 3-bar window each bar

    c1_idx = bar_index - 2

    c2_idx = bar_index - 1

    // c2 must be within CISD range

    if c2_idx >= cisd_start_idx and c2_idx <= cisd_end_idx + 5

        if bias == 1

            // Bullish FVG: high[2] < low[0] (gap between candle1 high and candle3 low)

            // candle1 high = fvg_low, candle3 low = fvg_high (wait — bullish FVG:)

            // c1 high = FVG low, c3 low = FVG high → price trades UP through the gap

            // BUT this is a BEARISH FVG we need inside a bearish CISD range for LONG

            // Per spec: bearish FVG: c1 low = FVG high, c3 high = FVG low

            // Candle 2 is displacement DOWN, gap between c1 low and c3 high

            fvg_h = low[2]   // candle1 low

            fvg_l = high[0]  // candle3 high

            if fvg_h > fvg_l  // valid gap

                fvg_high  := fvg_h

                fvg_low   := fvg_l

                fvg_valid := true

  

        else if bias == -1

            // Bullish FVG inside bullish CISD range for SHORT

            // c1 high = FVG low, c3 low = FVG high

            fvg_l = high[2]  // candle1 high

            fvg_h = low[0]   // candle3 low

            if fvg_h > fvg_l

                fvg_high  := fvg_h

                fvg_low   := fvg_l

                fvg_valid := true

  

// ============================================================

// ENTRY LOGIC

// CISD confirmation = close breaks above CISD range high (long)

//                   = close breaks below CISD range low (short)

// ============================================================

long_cisd_confirm  = bias == 1 and not na(cisd_range_high) and close > cisd_range_high and fvg_valid and not in_trade and not setup_used and in_entry_session

short_cisd_confirm = bias == -1 and not na(cisd_range_low) and close < cisd_range_low and fvg_valid and not in_trade and not setup_used and in_entry_session

  

// ---- LONG ENTRY ----

if long_cisd_confirm

    entry_price := close

    sl_price    := cisd_sl_level           // lowest close in CISD range

    tp_price    := entry_price + (entry_price - sl_price)  // 1RR

  

    strategy.entry("Long", strategy.long, comment="CISD+FVG Long")

    strategy.exit("Long Exit", "Long", stop=sl_price, limit=tp_price)

  

    in_trade    := true

    setup_used  := true

    bias        := 0

  

    // Draw trade levels

    line.delete(sl_line)

    line.delete(tp_line)

    sl_line := line.new(bar_index, sl_price, bar_index + 50, sl_price, color=color.red, style=line.style_solid, width=1)

    tp_line := line.new(bar_index, tp_price, bar_index + 50, tp_price, color=color.green, style=line.style_solid, width=1)

  

    // Draw CISD confirmation line (green)

    line.delete(cisd_line)

    cisd_line := line.new(cisd_start_idx, cisd_range_high, bar_index + 10, cisd_range_high, color=color.green, style=line.style_solid, width=2)

  

    // Entry arrow

    label.new(bar_index, low - syminfo.mintick * 5, "▲", style=label.style_none, textcolor=color.green, size=size.small)

  

// ---- SHORT ENTRY ----

if short_cisd_confirm

    entry_price := close

    sl_price    := cisd_sl_level           // highest close in CISD range

    tp_price    := entry_price - (sl_price - entry_price)  // 1RR

  

    strategy.entry("Short", strategy.short, comment="CISD+FVG Short")

    strategy.exit("Short Exit", "Short", stop=sl_price, limit=tp_price)

  

    in_trade    := true

    setup_used  := true

    bias        := 0

  

    line.delete(sl_line)

    line.delete(tp_line)

    sl_line := line.new(bar_index, sl_price, bar_index + 50, sl_price, color=color.red, style=line.style_solid, width=1)

    tp_line := line.new(bar_index, tp_price, bar_index + 50, tp_price, color=color.green, style=line.style_solid, width=1)

  

    line.delete(cisd_line)

    cisd_line := line.new(cisd_start_idx, cisd_range_low, bar_index + 10, cisd_range_low, color=color.red, style=line.style_solid, width=2)

  

    label.new(bar_index, high + syminfo.mintick * 5, "▼", style=label.style_none, textcolor=color.red, size=size.small)

  

// ============================================================

// TRADE CLOSE DETECTION

// ============================================================

trade_closed = in_trade and strategy.position_size == 0

  

if trade_closed

    in_trade    := false

    setup_used  := false   // allow fresh setup after next swing break

    entry_price := na

    sl_price    := na

    tp_price    := na

    cisd_start_idx  := na

    cisd_end_idx    := na

    cisd_range_high := na

    cisd_range_low  := na

    cisd_sl_level   := na

    fvg_high        := na

    fvg_low         := na

    fvg_valid       := false

    bear_run_start  := na

    bear_run_count  := 0

    bull_run_start  := na

    bull_run_count  := 0

  

// ============================================================

// CISD RANGE BOUNDARY VISUAL (orange line = active range boundary)

// ============================================================

// Show orange line at the CISD range boundary while a valid range exists

show_cisd_boundary = not na(cisd_start_idx) and not in_trade and not setup_used

  

if show_cisd_boundary

    line.delete(cisd_range_line)

    if bias == 1

        cisd_range_line := line.new(cisd_start_idx, cisd_range_high, bar_index, cisd_range_high, color=color.orange, style=line.style_dashed, width=1)

    else if bias == -1

        cisd_range_line := line.new(cisd_start_idx, cisd_range_low, bar_index, cisd_range_low, color=color.orange, style=line.style_dashed, width=1)

  

// ============================================================

// LEGEND COMMENT

// ============================================================

// Red  dashed  = unbroken 5m swing high

// Green dashed = unbroken 5m swing low

// Red  dotted  = broken 5m swing high (bearish bias trigger)

// Green dotted = broken 5m swing low (bullish bias trigger)

// Orange dashed = active CISD range boundary

// Green solid (thick) = bullish CISD confirmation level

// Red solid (thick)   = bearish CISD confirmation level

// Red solid (thin)    = stop loss

// Green solid (thin)  = take profit

// ▲ = long entry | ▼ = short entry
```
