
//@version=5
indicator("Swing - First FVG Only", overlay=true, max_boxes_count=500, max_lines_count=500)

// TIMEFRAME CHECK
is_1m = timeframe.period == "1"

// 5M SWINGS
[h5, l5]     = request.security(syminfo.tickerid, "5", [high,    low],    lookahead=barmerge.lookahead_off)
[h5_1, l5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1]], lookahead=barmerge.lookahead_off)
[h5_2, l5_2] = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

is_swing_high = h5_1 > h5_2 and h5_1 > h5
is_swing_low  = l5_1 < l5_2 and l5_1 < l5

// STATE
var float recentHigh  = na
var float recentLow   = na
var float brokenLevel = na
var int   highBar     = na
var int   lowBar      = na
var line  highLine    = na
var line  lowLine     = na
var line  brokenLine  = na
var box   fvgBox      = na
var int   mode        = 0
var bool  fvgFound    = false

// UPDATE SWING LINES
if is_swing_high
    recentHigh := h5_1
    highBar    := bar_index - 1
    if not na(highLine)
        line.delete(highLine)
    highLine := line.new(x1=bar_index-1, y1=recentHigh, x2=bar_index+300, y2=recentHigh, color=color.red, width=2, style=line.style_dashed, xloc=xloc.bar_index)

if is_swing_low
    recentLow := l5_1
    lowBar    := bar_index - 1
    if not na(lowLine)
        line.delete(lowLine)
    lowLine := line.new(x1=bar_index-1, y1=recentLow, x2=bar_index+300, y2=recentLow, color=color.green, width=2, style=line.style_dashed, xloc=xloc.bar_index)

// BREAK DETECTION
if not na(recentHigh) and high > recentHigh
    mode        := -1
    fvgFound    := false
    brokenLevel := recentHigh
    recentHigh  := na
    if not na(fvgBox)
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=highBar, y1=brokenLevel, x2=bar_index+300, y2=brokenLevel, color=color.red, width=2, style=line.style_dotted, xloc=xloc.bar_index)

if not na(recentLow) and low < recentLow
    mode        := 1
    fvgFound    := false
    brokenLevel := recentLow
    recentLow   := na
    if not na(fvgBox)
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=lowBar, y1=brokenLevel, x2=bar_index+300, y2=brokenLevel, color=color.green, width=2, style=line.style_dotted, xloc=xloc.bar_index)

// FIRST FVG AFTER BREAK - 1m only
bull_fvg = low > high[2]
bear_fvg = high < low[2]

if is_1m and mode != 0 and not fvgFound
    if mode == 1 and bull_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox   := box.new(bar_index[2], low, bar_index, high[2], bgcolor=color.new(color.green, 70), border_color=color.green)
        fvgFound := true

    if mode == -1 and bear_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox   := box.new(bar_index[2], low[2], bar_index, high, bgcolor=color.new(color.red, 70), border_color=color.red)
        fvgFound := true

// STRETCH FVG BOX
if not na(fvgBox) and fvgFound
    box.set_right(fvgBox, bar_index)
