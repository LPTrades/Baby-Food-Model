//@version=5
indicator("Swing - First FVG Only", overlay=true, max_boxes_count=500, max_lines_count=500, max_labels_count=500)

// TIMEFRAME CHECK
is_1m = timeframe.period == "1"

// 5M SWINGS
[h5, l5]     = request.security(syminfo.tickerid, "5", [high,    low],    lookahead=barmerge.lookahead_off)
[h5_1, l5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1]], lookahead=barmerge.lookahead_off)
[h5_2, l5_2] = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

is_swing_high = h5_1 > h5_2 and h5_1 > h5
is_swing_low  = l5_1 < l5_2 and l5_1 < l5

// 5M TIMESTAMPS FOR ANCHORING
[t5_1] = request.security(syminfo.tickerid, "5", [time[1]], lookahead=barmerge.lookahead_off)

// STATE
var float recentHigh  = na
var float recentLow   = na
var float brokenLevel = na
var int   highTime    = na
var int   lowTime     = na
var line  highLine    = na
var line  lowLine     = na
var line  brokenLine  = na
var box   fvgBox      = na
var box   stopBox     = na
var box   tpBox       = na
var int   mode        = 0
var int   tradeMode   = 0
var bool  fvgFound    = false
var float entryLevel  = na
var bool  entryHit    = false
var int   fvgBar      = na
var float structStop  = na
var float tpLevel     = na
var float slLevel     = na
var bool  tradeOpen   = false

// UPDATE SWING LINES
if is_swing_high
    recentHigh := h5_1
    highTime   := t5_1
    if not na(highLine)
        line.delete(highLine)
    highLine := line.new(x1=highTime, y1=recentHigh, x2=highTime + 300 * 60000, y2=recentHigh, color=color.red, width=2, style=line.style_dashed, xloc=xloc.bar_time)

if is_swing_low
    recentLow := l5_1
    lowTime   := t5_1
    if not na(lowLine)
        line.delete(lowLine)
    lowLine := line.new(x1=lowTime, y1=recentLow, x2=lowTime + 300 * 60000, y2=recentLow, color=color.green, width=2, style=line.style_dashed, xloc=xloc.bar_time)

// BREAK DETECTION
if not na(recentHigh) and high > recentHigh
    mode        := -1
    fvgFound    := false
    brokenLevel := recentHigh
    recentHigh  := na
    entryLevel  := na
    entryHit    := tradeOpen ? entryHit : false
    fvgBar      := na
    structStop  := na
    if not na(fvgBox) and not tradeOpen
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=highTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.red, width=2, style=line.style_dotted, xloc=xloc.bar_time)

if not na(recentLow) and low < recentLow
    mode        := 1
    fvgFound    := false
    brokenLevel := recentLow
    recentLow   := na
    entryLevel  := na
    entryHit    := tradeOpen ? entryHit : false
    fvgBar      := na
    structStop  := na
    if not na(fvgBox) and not tradeOpen
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=lowTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.green, width=2, style=line.style_dotted, xloc=xloc.bar_time)

// TRACK STRUCTURAL STOP AFTER BREAK
if mode == -1 and not fvgFound
    structStop := na(structStop) ? high : math.max(structStop, high)

if mode == 1 and not fvgFound
    structStop := na(structStop) ? low : math.min(structStop, low)

// FIRST FVG AFTER BREAK - 1m only
bull_fvg = low > high[2]
bear_fvg = high < low[2]

if is_1m and mode != 0 and not fvgFound

    if mode == 1 and bull_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox     := box.new(left=bar_index[2], top=low, right=bar_index, bottom=high[2], bgcolor=color.new(color.green, 70), border_color=color.green)
        fvgFound   := true
        entryLevel := low
        entryHit   := false
        fvgBar     := bar_index

    if mode == -1 and bear_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox     := box.new(left=bar_index[2], top=low[2], right=bar_index, bottom=high, bgcolor=color.new(color.red, 70), border_color=color.red)
        fvgFound   := true
        entryLevel := high
        entryHit   := false
        fvgBar     := bar_index

// STRETCH FVG BOX
if not na(fvgBox) and fvgFound and not tradeOpen
    box.set_right(fvgBox, bar_index)

// STRETCH STOP AND TP BOXES
if tradeOpen
    if not na(stopBox)
        box.set_right(stopBox, bar_index + 100)
    if not na(tpBox)
        box.set_right(tpBox, bar_index + 100)

// ENTRY TRIGGER
if is_1m and fvgFound and not entryHit and not tradeOpen and not na(entryLevel) and not na(fvgBar) and bar_index > fvgBar

    if mode == 1 and low <= entryLevel
        entryHit   := true
        tradeOpen  := true
        tradeMode  := 1
        risk        = math.abs(entryLevel - structStop)
        tpLevel    := entryLevel + 2 * risk
        slLevel    := structStop
        label.new(bar_index, low, text="▲", style=label.style_label_up, color=color.green, textcolor=color.white, size=size.normal)
        if not na(stopBox)
            box.delete(stopBox)
        if not na(tpBox)
            box.delete(tpBox)
        stopBox := box.new(left=bar_index, top=entryLevel, right=bar_index + 100, bottom=slLevel,    bgcolor=color.new(color.red, 80),   border_color=color.red)
        tpBox   := box.new(left=bar_index, top=tpLevel,    right=bar_index + 100, bottom=entryLevel, bgcolor=color.new(color.green, 80), border_color=color.green)

    if mode == -1 and high >= entryLevel
        entryHit   := true
        tradeOpen  := true
        tradeMode  := -1
        risk        = math.abs(entryLevel - structStop)
        tpLevel    := entryLevel - 2 * risk
        slLevel    := structStop
        label.new(bar_index, high, text="▼", style=label.style_label_down, color=color.red, textcolor=color.white, size=size.normal)
        if not na(stopBox)
            box.delete(stopBox)
        if not na(tpBox)
            box.delete(tpBox)
        stopBox := box.new(left=bar_index, top=slLevel,    right=bar_index + 100, bottom=entryLevel, bgcolor=color.new(color.red, 80),   border_color=color.red)
        tpBox   := box.new(left=bar_index, top=entryLevel, right=bar_index + 100, bottom=tpLevel,    bgcolor=color.new(color.green, 80), border_color=color.green)

// CLOSE TRADE - hit TP
if tradeOpen and not na(tpLevel)
    if tradeMode == 1 and high >= tpLevel
        if not na(stopBox)
            box.delete(stopBox)
            stopBox := na
        if not na(tpBox)
            box.delete(tpBox)
            tpBox := na
        if not na(fvgBox)
            box.delete(fvgBox)
            fvgBox := na
        tradeOpen := false
        tradeMode := 0
        tpLevel   := na
        slLevel   := na

    if tradeMode == -1 and low <= tpLevel
        if not na(stopBox)
            box.delete(stopBox)
            stopBox := na
        if not na(tpBox)
            box.delete(tpBox)
            tpBox := na
        if not na(fvgBox)
            box.delete(fvgBox)
            fvgBox := na
        tradeOpen := false
        tradeMode := 0
        tpLevel   := na
        slLevel   := na

// CLOSE TRADE - hit SL
if tradeOpen and not na(slLevel)
    if tradeMode == 1 and low <= slLevel
        if not na(stopBox)
            box.delete(stopBox)
            stopBox := na
        if not na(tpBox)
            box.delete(tpBox)
            tpBox := na
        if not na(fvgBox)
            box.delete(fvgBox)
            fvgBox := na
        tradeOpen := false
        tradeMode := 0
        tpLevel   := na
        slLevel   := na

    if tradeMode == -1 and high >= slLevel
        if not na(stopBox)
            box.delete(stopBox)
            stopBox := na
        if not na(tpBox)
            box.delete(tpBox)
            tpBox := na
        if not na(fvgBox)
            box.delete(fvgBox)
            fvgBox := na
        tradeOpen := false
        tradeMode := 0
        tpLevel   := na
        slLevel   := na
