//@version=5

strategy("5m Break → 1m CISD + FVG Strategy", overlay=true, max_lines_count=200, max_labels_count=200, default_qty_type=strategy.fixed, default_qty_value=1)

// ============================================================
// SETTINGS
// ============================================================
session_input    = input.session("0930-1100", "Entry Session (ET)")
session_full     = input.session("0930-1200", "Full Session (ET)")

// ============================================================
// SESSION HELPERS
// ============================================================
in_entry_session = not na(time("1", session_input, "America/New_York"))
in_full_session  = not na(time("1", session_full,  "America/New_York"))
session_end_time = timestamp("America/New_York", year, month, dayofmonth, 12, 0, 0)

// ============================================================
// 5m DATA
// ============================================================
[h5,  l5 ]         = request.security(syminfo.tickerid, "5", [high,    low   ], lookahead=barmerge.lookahead_off)
[h5_1, l5_1, t5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1], time[1]], lookahead=barmerge.lookahead_off)
[h5_2, l5_2]       = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

is_swing_high_5m = h5_1 > h5_2 and h5_1 > h5
is_swing_low_5m  = l5_1 < l5_2 and l5_1 < l5

new_5m_bar = ta.change(t5_1) != 0

new_session_1m = in_full_session and not in_full_session[1]

// ============================================================
// 5m SWING STATE
// ============================================================
var float recent_sh      = na
var float recent_sl      = na
var int   recent_sh_time = na
var int   recent_sl_time = na

var line  recent_sh_line = na
var line  recent_sl_line = na
var line  broken_sh_line = na
var line  broken_sl_line = na

// ============================================================
// STRATEGY STATE
// ============================================================
var int   bias       = 0
var bool  setup_used = false

// CISD — bullish
var int   bear_run_start  = na
var int   bear_run_count  = 0
var float bear_run_open   = na
var float bear_run_sl     = na

var int   cisd_long_start = na
var int   cisd_long_end   = na
var float cisd_long_high  = na
var float cisd_long_low   = na
var float cisd_long_sl    = na

// CISD — bearish
var int   bull_run_start  = na
var int   bull_run_count  = 0
var float bull_run_open   = na
var float bull_run_sl     = na

var int   cisd_short_start = na
var int   cisd_short_end   = na
var float cisd_short_high  = na
var float cisd_short_low   = na
var float cisd_short_sl    = na

// FVG — directional
var float fvg_long_high   = na
var float fvg_long_low    = na
var bool  fvg_long_valid  = false

var float fvg_short_high  = na
var float fvg_short_low   = na
var bool  fvg_short_valid = false

// Trade levels
var float entry_price = na
var float sl_price    = na
var float tp_price    = na
var bool  in_trade    = false

var line  sl_line         = na
var line  tp_line         = na
var line  cisd_line       = na
var line  cisd_range_line = na

// ============================================================
// RESET ON NEW SESSION
// ============================================================
if new_session_1m
    recent_sh      := na
    recent_sl      := na
    recent_sh_time := na
    recent_sl_time := na
    line.delete(recent_sh_line)
    line.delete(recent_sl_line)
    line.delete(broken_sh_line)
    line.delete(broken_sl_line)
    recent_sh_line   := na
    recent_sl_line   := na
    broken_sh_line   := na
    broken_sl_line   := na
    bias             := 0
    setup_used       := false
    bear_run_start   := na
    bear_run_count   := 0
    bear_run_open    := na
    bear_run_sl      := na
    cisd_long_start  := na
    cisd_long_end    := na
    cisd_long_high   := na
    cisd_long_low    := na
    cisd_long_sl     := na
    bull_run_start   := na
    bull_run_count   := 0
    bull_run_open    := na
    bull_run_sl      := na
    cisd_short_start := na
    cisd_short_end   := na
    cisd_short_high  := na
    cisd_short_low   := na
    cisd_short_sl    := na
    fvg_long_high    := na
    fvg_long_low     := na
    fvg_long_valid   := false
    fvg_short_high   := na
    fvg_short_low    := na
    fvg_short_valid  := false

// ============================================================
// 5m SWING CAPTURE
// Reset run tracker only — keep stored CISD range
// ============================================================
if in_full_session and new_5m_bar and is_swing_high_5m
    recent_sh      := h5_1
    recent_sh_time := t5_1
    bull_run_start := na
    bull_run_count := 0
    bull_run_open  := na
    bull_run_sl    := na
    line.delete(recent_sh_line)
    recent_sh_line := line.new(recent_sh_time, recent_sh, session_end_time, recent_sh, xloc=xloc.bar_time, color=color.red, style=line.style_dashed, width=1)

if in_full_session and new_5m_bar and is_swing_low_5m
    recent_sl      := l5_1
    recent_sl_time := t5_1
    bear_run_start := na
    bear_run_count := 0
    bear_run_open  := na
    bear_run_sl    := na
    line.delete(recent_sl_line)
    recent_sl_line := line.new(recent_sl_time, recent_sl, session_end_time, recent_sl, xloc=xloc.bar_time, color=color.green, style=line.style_dashed, width=1)

// ============================================================
// CONTINUOUS CISD TRACKING + FVG SCAN
// Gate: swing exists OR bias set — tracks before and after break
// CISD only updates on new extreme — FVG persists unless range updates
// ============================================================
is_bearish_1m = close < open
is_bullish_1m = close > open

// ----- BULLISH CISD + FVG (longs) -----
if in_entry_session and (not na(recent_sl) or bias == 1) and not in_trade and not setup_used
    if is_bearish_1m
        if na(bear_run_start) or not is_bearish_1m[1]
            // New consecutive bearish sequence starting
            // Only replace stored CISD if this sequence makes a new lower low
            bear_run_start := bar_index
            bear_run_open  := open
            bear_run_sl    := close
            bear_run_count := 1
            // Do NOT wipe cisd_long or fvg here yet — wait to see if new extreme forms
        else
            // Extend existing run
            bear_run_count += 1
            bear_run_sl    := math.min(bear_run_sl, close)

        // Capture or update CISD only if this run makes a new lower low
        // (or no range exists yet)
        if bear_run_count >= 2
            if na(cisd_long_low) or low < cisd_long_low
                // New extreme — update range and wipe FVG
                cisd_long_start := bear_run_start
                cisd_long_end   := bar_index
                cisd_long_high  := bear_run_open   // locked to first candle open
                cisd_long_low   := low
                cisd_long_sl    := bear_run_sl
                fvg_long_high   := na
                fvg_long_low    := na
                fvg_long_valid  := false
            else
                // No new extreme — update end/sl only, keep level and FVG
                cisd_long_end := bar_index
                cisd_long_sl  := math.min(cisd_long_sl, bear_run_sl)
    else
        // Non-bearish — reset run tracker only, keep range and FVG
        bear_run_start := na
        bear_run_count := 0
        bear_run_open  := na
        bear_run_sl    := na

    // FVG scan — runs every bar once CISD exists, persists until range makes new extreme
    if not fvg_long_valid and not na(cisd_long_start) and bar_index >= 2
        c2_idx = bar_index - 1
        if c2_idx >= cisd_long_start
            fvg_h = low[2]
            fvg_l = high[0]
            if fvg_h > fvg_l
                fvg_long_high  := fvg_h
                fvg_long_low   := fvg_l
                fvg_long_valid := true

// ----- BEARISH CISD + FVG (shorts) -----
if in_entry_session and (not na(recent_sh) or bias == -1) and not in_trade and not setup_used
    if is_bullish_1m
        if na(bull_run_start) or not is_bullish_1m[1]
            bull_run_start := bar_index
            bull_run_open  := open
            bull_run_sl    := close
            bull_run_count := 1
        else
            bull_run_count += 1
            bull_run_sl    := math.max(bull_run_sl, close)

        if bull_run_count >= 2
            if na(cisd_short_high) or high > cisd_short_high
                // New extreme — update range and wipe FVG
                cisd_short_start := bull_run_start
                cisd_short_end   := bar_index
                cisd_short_low   := bull_run_open  // locked to first candle open
                cisd_short_high  := high
                cisd_short_sl    := bull_run_sl
                fvg_short_high   := na
                fvg_short_low    := na
                fvg_short_valid  := false
            else
                // No new extreme — update end/sl only, keep level and FVG
                cisd_short_end := bar_index
                cisd_short_sl  := math.max(cisd_short_sl, bull_run_sl)
    else
        bull_run_start := na
        bull_run_count := 0
        bull_run_open  := na
        bull_run_sl    := na

    if not fvg_short_valid and not na(cisd_short_start) and bar_index >= 2
        c2_idx = bar_index - 1
        if c2_idx >= cisd_short_start
            fvg_l = high[2]
            fvg_h = low[0]
            if fvg_h > fvg_l
                fvg_short_high  := fvg_h
                fvg_short_low   := fvg_l
                fvg_short_valid := true

// ============================================================
// 5m BREAK DETECTION → SET BIAS
// ============================================================
if in_full_session and not na(recent_sh) and high > recent_sh and not in_trade and not setup_used
    line.delete(broken_sh_line)
    broken_sh_line := line.new(recent_sh_time, recent_sh, session_end_time, recent_sh, xloc=xloc.bar_time, color=color.red, style=line.style_dotted, width=1)
    line.delete(recent_sh_line)
    recent_sh_line := na
    recent_sh      := na
    bias           := -1
    setup_used     := false

if in_full_session and not na(recent_sl) and low < recent_sl and not in_trade and not setup_used
    line.delete(broken_sl_line)
    broken_sl_line := line.new(recent_sl_time, recent_sl, session_end_time, recent_sl, xloc=xloc.bar_time, color=color.green, style=line.style_dotted, width=1)
    line.delete(recent_sl_line)
    recent_sl_line := na
    recent_sl      := na
    bias           := 1
    setup_used     := false

// ============================================================
// ENTRY LOGIC
// ============================================================
long_cisd_confirm  = bias == 1  and not na(cisd_long_high)  and close > cisd_long_high  and fvg_long_valid  and not in_trade and not setup_used and in_entry_session
short_cisd_confirm = bias == -1 and not na(cisd_short_low)  and close < cisd_short_low  and fvg_short_valid and not in_trade and not setup_used and in_entry_session

if long_cisd_confirm
    entry_price := close
    sl_price    := cisd_long_sl
    tp_price    := entry_price + (entry_price - sl_price)

    strategy.entry("Long", strategy.long, comment="CISD+FVG Long")
    strategy.exit("Long Exit", "Long", stop=sl_price, limit=tp_price)

    in_trade   := true
    setup_used := true
    bias       := 0

    line.delete(sl_line)
    line.delete(tp_line)
    sl_line := line.new(bar_index, sl_price, bar_index + 50, sl_price, color=color.red,   style=line.style_solid, width=1)
    tp_line := line.new(bar_index, tp_price, bar_index + 50, tp_price, color=color.green, style=line.style_solid, width=1)
    line.delete(cisd_line)
    cisd_line := line.new(cisd_long_start, cisd_long_high, bar_index + 10, cisd_long_high, color=color.green, style=line.style_solid, width=2)
    label.new(bar_index, low - syminfo.mintick * 5, "▲", style=label.style_none, textcolor=color.green, size=size.small)

if short_cisd_confirm
    entry_price := close
    sl_price    := cisd_short_sl
    tp_price    := entry_price - (sl_price - entry_price)

    strategy.entry("Short", strategy.short, comment="CISD+FVG Short")
    strategy.exit("Short Exit", "Short", stop=sl_price, limit=tp_price)

    in_trade   := true
    setup_used := true
    bias       := 0

    line.delete(sl_line)
    line.delete(tp_line)
    sl_line := line.new(bar_index, sl_price, bar_index + 50, sl_price, color=color.red,   style=line.style_solid, width=1)
    tp_line := line.new(bar_index, tp_price, bar_index + 50, tp_price, color=color.green, style=line.style_solid, width=1)
    line.delete(cisd_line)
    cisd_line := line.new(cisd_short_start, cisd_short_low, bar_index + 10, cisd_short_low, color=color.red, style=line.style_solid, width=2)
    label.new(bar_index, high + syminfo.mintick * 5, "▼", style=label.style_none, textcolor=color.red, size=size.small)

// ============================================================
// TRADE CLOSE DETECTION
// ============================================================
trade_closed = in_trade and strategy.position_size == 0

if trade_closed
    in_trade         := false
    setup_used       := false
    entry_price      := na
    sl_price         := na
    tp_price         := na
    cisd_long_start  := na
    cisd_long_end    := na
    cisd_long_high   := na
    cisd_long_low    := na
    cisd_long_sl     := na
    cisd_short_start := na
    cisd_short_end   := na
    cisd_short_high  := na
    cisd_short_low   := na
    cisd_short_sl    := na
    fvg_long_high    := na
    fvg_long_low     := na
    fvg_long_valid   := false
    fvg_short_high   := na
    fvg_short_low    := na
    fvg_short_valid  := false
    bear_run_start   := na
    bear_run_count   := 0
    bull_run_start   := na
    bull_run_count   := 0

// ============================================================
// CISD RANGE BOUNDARY VISUAL
// ============================================================
if bias == 1 and not na(cisd_long_start) and not in_trade
    line.delete(cisd_range_line)
    cisd_range_line := line.new(cisd_long_start, cisd_long_high, bar_index, cisd_long_high, color=color.orange, style=line.style_dashed, width=1)
else if bias == -1 and not na(cisd_short_start) and not in_trade
    line.delete(cisd_range_line)
    cisd_range_line := line.new(cisd_short_start, cisd_short_low, bar_index, cisd_short_low, color=color.orange, style=line.style_dashed, width=1)

// ============================================================
// LEGEND
// ============================================================
// Red  dashed  = unbroken 5m swing high
// Green dashed = unbroken 5m swing low
// Red  dotted  = broken 5m swing high (bearish bias)
// Green dotted = broken 5m swing low  (bullish bias)
// Orange dashed = active CISD range boundary
// Green solid thick = bullish CISD confirmation level
// Red solid thick   = bearish CISD confirmation level
// Red solid thin    = stop loss
// Green solid thin  = take profit
// ▲ = long entry | ▼ = short entry
