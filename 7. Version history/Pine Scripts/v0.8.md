
# Version Update - 2026-05-24

## Changes

- 6-candle body stop loss attached to the entry signal.
- Stop = lowest open-or-close across all 6 candles for a long (highest for a short): FVG C1/C2/C3 + IFVG C1/C2/C3. Bodies only — wicks ignored.
- Captured the 6 candle bodies at detection time: FVG bodies stored when the FVG locks, IFVG bodies stored when the originating opposite gap is tracked (before the flip).
- 1.5R take profit attached. R = `|entry - stop|`. Long TP = `entry + 1.5 * R`, short TP = `entry - 1.5 * R`. Configurable via `tpRR`.
- Entry, SL, and TP now drawn as lines from the entry bar (blue solid entry, orange dashed SL, teal dashed TP).
- Added state vars: `entryPrice`, `slPrice`, `tpPrice`, plus the 12 body-capture floats (`fvg_c1_o`...`ifvg_c3_c`) and the `slLine` / `tpLine` / `entryLine` line handles.
- `slPrice` and `tpPrice` added to both break-detection reset blocks so they clear on each new sweep.

## Bugs Fixed

- Comma-joined assignments (`a := x, b := y` on one line) caused a compile error: "Value with NA type cannot be assigned to a variable that was defined without type keyword" at the body-capture lines. Split every multi-assignment onto its own line. Pine does not support that comma syntax.

## Reason For Changes

- Move from signal-only (v0.8) to a fully defined trade: entry, stop, and target all set the moment the entry fires.
- Body-based stop (open/close, no wicks) was the explicit choice over the old displacement-leg extreme — it ties risk to the structure of the two gaps that actually produced the setup rather than the full leg.
- All 6 candles pooled (both gaps) so the stop sits beyond whichever gap extends furthest, giving one consistent risk boundary regardless of which gap confirmed last.
- 1.5R target chosen over the old 2R from The Oracle spec.

## Observations

- Still visualization-only for trade management — the script draws entry/SL/TP levels but does NOT determine which is hit first, and does not block a new setup while one is live. That win/loss resolution is the backtester's job.
- Stop distance now varies with gap size. Tight gaps produce tight stops and small absolute TP distances; wide gaps the reverse. Worth watching whether very tight body-stops get wicked out at a high rate.
- Entry still fills at the open of the candle after both confirm (carried from v0.8), so the one-bar-delay slippage note from v0.8 still applies.
- Detectors remain order-agnostic (FVG-then-IFVG or IFVG-then-FVG both accepted) — unchanged from v0.8.

```
//@version=5
indicator("Swing - FVG + IFVG + Entry + SL + TP", overlay=true, max_boxes_count=500, max_lines_count=500, max_labels_count=500)

// CONFIG
tpRR = 1.5   // risk:reward target

// TIMEFRAME CHECK
is_1m = timeframe.period == "1"

// 5M SWINGS
[h5, l5]     = request.security(syminfo.tickerid, "5", [high,    low],    lookahead=barmerge.lookahead_off)
[h5_1, l5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1]], lookahead=barmerge.lookahead_off)
[h5_2, l5_2] = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

is_swing_high = h5_1 > h5_2 and h5_1 > h5
is_swing_low  = l5_1 < l5_2 and l5_1 < l5

// 5M TIMESTAMPS FOR ANCHORING
[t5_1] = request.security(syminfo.tickerid, "5", [time[1]], lookahead=barmerge.lookahead_off)

// STATE
var float recentHigh   = na
var float recentLow    = na
var float brokenLevel  = na
var int   highTime     = na
var int   lowTime      = na
var line  highLine     = na
var line  lowLine      = na
var line  brokenLine   = na
var box   ifvgBox      = na
var box   fvgBox       = na
var int   mode         = 0

// IFVG detector state
var bool  ifvgFound    = false
var bool  pendingGap   = false
var float gapEdge      = na
var int   gapLeft      = na
var float gapTop       = na
var float gapBot       = na

// FVG detector state
var bool  fvgFound     = false

// CANDLE BODY CAPTURE for stop calc
var float fvg_c1_o = na
var float fvg_c1_c = na
var float fvg_c2_o = na
var float fvg_c2_c = na
var float fvg_c3_o = na
var float fvg_c3_c = na
var float ifvg_c1_o = na
var float ifvg_c1_c = na
var float ifvg_c2_o = na
var float ifvg_c2_c = na
var float ifvg_c3_o = na
var float ifvg_c3_c = na

// ENTRY state
var bool  entryArmed   = false
var bool  entryFired   = false

// TRADE state
var float entryPrice   = na
var float slPrice      = na
var float tpPrice      = na
var line  slLine       = na
var line  tpLine       = na
var line  entryLine    = na

// UPDATE SWING LINES
if is_swing_high
    recentHigh := h5_1
    highTime   := t5_1
    if not na(highLine)
        line.delete(highLine)
    highLine := line.new(x1=highTime, y1=recentHigh, x2=highTime + 300 * 60000, y2=recentHigh, color=color.red, width=2, style=line.style_dashed, xloc=xloc.bar_time)

if is_swing_low
    recentLow := l5_1
    lowTime   := t5_1
    if not na(lowLine)
        line.delete(lowLine)
    lowLine := line.new(x1=lowTime, y1=recentLow, x2=lowTime + 300 * 60000, y2=recentLow, color=color.green, width=2, style=line.style_dashed, xloc=xloc.bar_time)

// BREAK DETECTION
if not na(recentHigh) and high > recentHigh
    mode        := -1
    ifvgFound   := false
    pendingGap  := false
    gapEdge     := na
    fvgFound    := false
    entryArmed  := false
    entryFired  := false
    entryPrice  := na
    slPrice     := na
    tpPrice     := na
    brokenLevel := recentHigh
    recentHigh  := na
    if not na(ifvgBox)
        box.delete(ifvgBox)
        ifvgBox := na
    if not na(fvgBox)
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=highTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.red, width=2, style=line.style_dotted, xloc=xloc.bar_time)

if not na(recentLow) and low < recentLow
    mode        := 1
    ifvgFound   := false
    pendingGap  := false
    gapEdge     := na
    fvgFound    := false
    entryArmed  := false
    entryFired  := false
    entryPrice  := na
    slPrice     := na
    tpPrice     := na
    brokenLevel := recentLow
    recentLow   := na
    if not na(ifvgBox)
        box.delete(ifvgBox)
        ifvgBox := na
    if not na(fvgBox)
        box.delete(fvgBox)
        fvgBox := na
    if not na(brokenLine)
        line.delete(brokenLine)
    brokenLine := line.new(x1=lowTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.green, width=2, style=line.style_dotted, xloc=xloc.bar_time)

// GAP DETECTION - 1m only
bull_fvg = low > high[2]
bear_fvg = high < low[2]

// ===================== IFVG DETECTOR =====================
if is_1m and pendingGap and not ifvgFound and not na(gapEdge)
    if mode == 1 and close > gapEdge
        ifvgFound  := true
        pendingGap := false
        if not na(ifvgBox)
            box.delete(ifvgBox)
        ifvgBox := box.new(left=gapLeft, top=gapTop, right=bar_index, bottom=gapBot, bgcolor=color.new(color.green, 70), border_color=color.green, border_style=line.style_dashed)
    if mode == -1 and close < gapEdge
        ifvgFound  := true
        pendingGap := false
        if not na(ifvgBox)
            box.delete(ifvgBox)
        ifvgBox := box.new(left=gapLeft, top=gapTop, right=bar_index, bottom=gapBot, bgcolor=color.new(color.red, 70), border_color=color.red, border_style=line.style_dashed)

if is_1m and mode != 0 and not ifvgFound
    if mode == 1 and bear_fvg
        pendingGap := true
        gapEdge    := low[2]
        gapTop     := low[2]
        gapBot     := high
        gapLeft    := bar_index[2]
        ifvg_c1_o := open[2]
        ifvg_c1_c := close[2]
        ifvg_c2_o := open[1]
        ifvg_c2_c := close[1]
        ifvg_c3_o := open
        ifvg_c3_c := close
    if mode == -1 and bull_fvg
        pendingGap := true
        gapEdge    := high[2]
        gapTop     := low
        gapBot     := high[2]
        gapLeft    := bar_index[2]
        ifvg_c1_o := open[2]
        ifvg_c1_c := close[2]
        ifvg_c2_o := open[1]
        ifvg_c2_c := close[1]
        ifvg_c3_o := open
        ifvg_c3_c := close

// ===================== FVG DETECTOR =====================
if is_1m and mode != 0 and not fvgFound
    if mode == 1 and bull_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox   := box.new(left=bar_index[2], top=low, right=bar_index, bottom=high[2], bgcolor=color.new(color.green, 85), border_color=color.green, border_style=line.style_solid)
        fvgFound := true
        fvg_c1_o := open[2]
        fvg_c1_c := close[2]
        fvg_c2_o := open[1]
        fvg_c2_c := close[1]
        fvg_c3_o := open
        fvg_c3_c := close
    if mode == -1 and bear_fvg
        if not na(fvgBox)
            box.delete(fvgBox)
        fvgBox   := box.new(left=bar_index[2], top=low[2], right=bar_index, bottom=high, bgcolor=color.new(color.red, 85), border_color=color.red, border_style=line.style_solid)
        fvgFound := true
        fvg_c1_o := open[2]
        fvg_c1_c := close[2]
        fvg_c2_o := open[1]
        fvg_c2_c := close[1]
        fvg_c3_o := open
        fvg_c3_c := close

// ===================== ENTRY TRIGGER + STOP + TP =====================
entrySignalLong  = false
entrySignalShort = false

if is_1m and not entryFired
    if entryArmed
        entryPrice := open
        if mode == 1
            entrySignalLong := true
            slPrice := math.min(math.min(math.min(fvg_c1_o, fvg_c1_c), math.min(fvg_c2_o, fvg_c2_c)), math.min(math.min(fvg_c3_o, fvg_c3_c), math.min(math.min(ifvg_c1_o, ifvg_c1_c), math.min(math.min(ifvg_c2_o, ifvg_c2_c), math.min(ifvg_c3_o, ifvg_c3_c)))))
            tpPrice := entryPrice + tpRR * (entryPrice - slPrice)
        if mode == -1
            entrySignalShort := true
            slPrice := math.max(math.max(math.max(fvg_c1_o, fvg_c1_c), math.max(fvg_c2_o, fvg_c2_c)), math.max(math.max(fvg_c3_o, fvg_c3_c), math.max(math.max(ifvg_c1_o, ifvg_c1_c), math.max(math.max(ifvg_c2_o, ifvg_c2_c), math.max(ifvg_c3_o, ifvg_c3_c)))))
            tpPrice := entryPrice - tpRR * (slPrice - entryPrice)
        entryArmed := false
        entryFired := true
        entryLine := line.new(x1=bar_index, y1=entryPrice, x2=bar_index + 60, y2=entryPrice, color=color.blue,   width=1, style=line.style_solid)
        slLine    := line.new(x1=bar_index, y1=slPrice,    x2=bar_index + 60, y2=slPrice,    color=color.orange, width=1, style=line.style_dashed)
        tpLine    := line.new(x1=bar_index, y1=tpPrice,    x2=bar_index + 60, y2=tpPrice,    color=color.teal,   width=1, style=line.style_dashed)
    else if fvgFound and ifvgFound
        entryArmed := true

// STRETCH BOXES
if not na(ifvgBox) and ifvgFound
    box.set_right(ifvgBox, bar_index)
if not na(fvgBox) and fvgFound
    box.set_right(fvgBox, bar_index)

// ===================== ARROWS =====================
plotshape(entrySignalLong,  title="Long Entry",  style=shape.triangleup,   location=location.belowbar, color=color.new(color.green, 0), size=size.small)
plotshape(entrySignalShort, title="Short Entry", style=shape.triangledown, location=location.abovebar, color=color.new(color.red, 0),   size=size.small)
```
