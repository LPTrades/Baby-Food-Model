# Version Update - 2026-05-23

## Changes Added

- [[Parallel FVG + IFVG detection after break]]

## Reason For Changes

- Run two independent gap detectors off the same break, in parallel: the v0.6 first-FVG detector and the IFVG flip detector.
- Each has its own state and its own box. Both reset on the next sweep.
- Detection + visualization only. No entry / SL / TP / trade management.

## FVG Logic (first continuation gap, v0.6 behavior)

- **Long** (`mode == 1`): first **bullish FVG** (`low > high[2]`). Locks on the first one, draws a solid-border box.
- **Short** (`mode == -1`): first **bearish FVG** (`high < low[2]`). Locks on the first one, solid-border box.

## IFVG Logic (last opposite gap, flips on close-through)

- **Long** (after swing low swept, `mode == 1`):
    - Track the **most recent** bearish FVG on 1m (`high < low[2]`), overwriting as newer ones form. The last one before the reversal is closest to price and the first to invert.
    - When a later bar **closes above** that tracked gap's top edge (`low[2]`), the bearish gap is invalidated and flips to a **bullish IFVG**. Draws a dashed-border box, sets `ifvgFound := true`.
- **Short** (after swing high swept, `mode == -1`):
    - Track the **most recent** bullish FVG on 1m (`low > high[2]`), overwriting as newer ones form.
    - When a later bar **closes below** that tracked gap's bottom edge (`high[2]`), the bullish gap flips to a **bearish IFVG**. Dashed-border box, sets `ifvgFound := true`.
- Flip check runs before the tracking update each bar, so a gap can't form and flip on the same bar.

## Telling the boxes apart

- **FVG**: solid border, lighter fill (85 transparency).
- **IFVG**: dashed border, heavier fill (70 transparency).

```
//@version=5

indicator("Swing - FVG + IFVG", overlay=true, max_boxes_count=500, max_lines_count=500, max_labels_count=500)

  

// TIMEFRAME CHECK

is_1m = timeframe.period == "1"

  

// 5M SWINGS

[h5, l5]     = request.security(syminfo.tickerid, "5", [high,    low],    lookahead=barmerge.lookahead_off)

[h5_1, l5_1] = request.security(syminfo.tickerid, "5", [high[1], low[1]], lookahead=barmerge.lookahead_off)

[h5_2, l5_2] = request.security(syminfo.tickerid, "5", [high[2], low[2]], lookahead=barmerge.lookahead_off)

  

is_swing_high = h5_1 > h5_2 and h5_1 > h5

is_swing_low  = l5_1 < l5_2 and l5_1 < l5

  

// 5M TIMESTAMPS FOR ANCHORING

[t5_1] = request.security(syminfo.tickerid, "5", [time[1]], lookahead=barmerge.lookahead_off)

  

// STATE

var float recentHigh   = na

var float recentLow    = na

var float brokenLevel  = na

var int   highTime     = na

var int   lowTime      = na

var line  highLine     = na

var line  lowLine      = na

var line  brokenLine   = na

var box   ifvgBox      = na

var box   fvgBox       = na

var int   mode         = 0

  

// IFVG detector state (last opposite gap, flips on close-through)

var bool  ifvgFound    = false   // true once IFVG is confirmed (flip happened)

var bool  pendingGap   = false   // true once an opposite-side gap is tracked, awaiting flip

var float gapEdge      = na      // the level price must close through to flip the gap

var int   gapLeft      = na      // bar_index of the gap's left edge (for box drawing)

var float gapTop       = na      // top of the original gap (for box drawing)

var float gapBot       = na      // bottom of the original gap (for box drawing)

  

// FVG detector state (first continuation gap, v0.6 behavior)

var bool  fvgFound     = false   // true once the first continuation FVG is found

  

// UPDATE SWING LINES

if is_swing_high

    recentHigh := h5_1

    highTime   := t5_1

    if not na(highLine)

        line.delete(highLine)

    highLine := line.new(x1=highTime, y1=recentHigh, x2=highTime + 300 * 60000, y2=recentHigh, color=color.red, width=2, style=line.style_dashed, xloc=xloc.bar_time)

  

if is_swing_low

    recentLow := l5_1

    lowTime   := t5_1

    if not na(lowLine)

        line.delete(lowLine)

    lowLine := line.new(x1=lowTime, y1=recentLow, x2=lowTime + 300 * 60000, y2=recentLow, color=color.green, width=2, style=line.style_dashed, xloc=xloc.bar_time)

  

// BREAK DETECTION

if not na(recentHigh) and high > recentHigh

    mode        := -1

    ifvgFound   := false

    pendingGap  := false

    gapEdge     := na

    fvgFound    := false

    brokenLevel := recentHigh

    recentHigh  := na

    if not na(ifvgBox)

        box.delete(ifvgBox)

        ifvgBox := na

    if not na(fvgBox)

        box.delete(fvgBox)

        fvgBox := na

    if not na(brokenLine)

        line.delete(brokenLine)

    brokenLine := line.new(x1=highTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.red, width=2, style=line.style_dotted, xloc=xloc.bar_time)

  

if not na(recentLow) and low < recentLow

    mode        := 1

    ifvgFound   := false

    pendingGap  := false

    gapEdge     := na

    fvgFound    := false

    brokenLevel := recentLow

    recentLow   := na

    if not na(ifvgBox)

        box.delete(ifvgBox)

        ifvgBox := na

    if not na(fvgBox)

        box.delete(fvgBox)

        fvgBox := na

    if not na(brokenLine)

        line.delete(brokenLine)

    brokenLine := line.new(x1=lowTime, y1=brokenLevel, x2=time + 300 * 60000, y2=brokenLevel, color=color.green, width=2, style=line.style_dotted, xloc=xloc.bar_time)

  

// GAP DETECTION - 1m only

bull_fvg = low > high[2]

bear_fvg = high < low[2]

  

// ===================== IFVG DETECTOR =====================

// IFVG STEP 1: CHECK FOR THE FLIP on the currently-tracked gap (close through the edge)

// Runs BEFORE the tracking update below so a gap can't form and flip on the same bar.

if is_1m and pendingGap and not ifvgFound and not na(gapEdge)

  

    // LONG flip: close ABOVE the bearish gap's top edge -> bullish IFVG

    if mode == 1 and close > gapEdge

        ifvgFound  := true

        pendingGap := false

        if not na(ifvgBox)

            box.delete(ifvgBox)

        ifvgBox := box.new(left=gapLeft, top=gapTop, right=bar_index, bottom=gapBot, bgcolor=color.new(color.green, 70), border_color=color.green, border_style=line.style_dashed)

  

    // SHORT flip: close BELOW the bullish gap's bottom edge -> bearish IFVG

    if mode == -1 and close < gapEdge

        ifvgFound  := true

        pendingGap := false

        if not na(ifvgBox)

            box.delete(ifvgBox)

        ifvgBox := box.new(left=gapLeft, top=gapTop, right=bar_index, bottom=gapBot, bgcolor=color.new(color.red, 70), border_color=color.red, border_style=line.style_dashed)

  

// IFVG STEP 2: TRACK THE MOST RECENT OPPOSITE-SIDE GAP (overwrite as new ones form)

// We keep the LAST opposite gap; it's the one closest to price and the first to invert.

if is_1m and mode != 0 and not ifvgFound

  

    // LONG: need a bearish gap to flip up

    if mode == 1 and bear_fvg

        pendingGap := true

        gapEdge    := low[2]      // top edge - price must close ABOVE this to flip

        gapTop     := low[2]

        gapBot     := high

        gapLeft    := bar_index[2]

  

    // SHORT: need a bullish gap to flip down

    if mode == -1 and bull_fvg

        pendingGap := true

        gapEdge    := high[2]     // bottom edge - price must close BELOW this to flip

        gapTop     := low

        gapBot     := high[2]

        gapLeft    := bar_index[2]

  

// ===================== FVG DETECTOR =====================

// First continuation gap after the break (v0.6 behavior). Locks on the FIRST one.

if is_1m and mode != 0 and not fvgFound

  

    // LONG: first bullish continuation gap

    if mode == 1 and bull_fvg

        if not na(fvgBox)

            box.delete(fvgBox)

        fvgBox   := box.new(left=bar_index[2], top=low, right=bar_index, bottom=high[2], bgcolor=color.new(color.green, 85), border_color=color.green, border_style=line.style_solid)

        fvgFound := true

  

    // SHORT: first bearish continuation gap

    if mode == -1 and bear_fvg

        if not na(fvgBox)

            box.delete(fvgBox)

        fvgBox   := box.new(left=bar_index[2], top=low[2], right=bar_index, bottom=high, bgcolor=color.new(color.red, 85), border_color=color.red, border_style=line.style_solid)

        fvgFound := true

  

// STRETCH BOXES

if not na(ifvgBox) and ifvgFound

    box.set_right(ifvgBox, bar_index)

if not na(fvgBox) and fvgFound

    box.set_right(fvgBox, bar_index)
```
