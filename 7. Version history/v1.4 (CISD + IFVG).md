def parse_session(s: str):
    """Return (start_time, end_time) as datetime.time objects."""
    m = re.fullmatch(r"(\d{1,2}):(\d{2})-(\d{1,2}):(\d{2})", s.strip())
    if not m:
        raise ValueError(f"Session format must be HH:MM-HH:MM, got: {s!r}")
    sh, sm, eh, em = map(int, m.groups())
    return dtime(sh, sm), dtime(eh, em)


def load_csv(path: str, freq_label: str) -> pd.DataFrame:
    """Load OHLC CSV, parse datetime, localize to ET, drop overnight candles."""
    print(f"  Loading {freq_label} data from: {path}")
    df = pd.read_csv(path, parse_dates=["datetime"])

    # Normalise column names to lower-case
    df.columns = [c.strip().lower() for c in df.columns]
    required = {"datetime", "open", "high", "low", "close"}
    missing  = required - set(df.columns)
    if missing:
        raise ValueError(f"CSV is missing columns: {missing}")

    # Timezone handling
    if df["datetime"].dt.tz is None:
        df["datetime"] = df["datetime"].dt.tz_localize("America/New_York", ambiguous="infer", nonexistent="shift_forward")
    else:
        df["datetime"] = df["datetime"].dt.tz_convert("America/New_York")

    df = df.sort_values("datetime").reset_index(drop=True)

    # Drop overnight candles (16:00 – 09:30)
    t = df["datetime"].dt.time
    overnight = (t >= NO_TRADE_START) | (t < NO_TRADE_END)
    dropped   = overnight.sum()
    df = df[~overnight].reset_index(drop=True)
    print(f"    {len(df):,} candles kept after dropping {dropped:,} overnight rows.")
    return df


def detect_5m_swings(df5: pd.DataFrame) -> pd.DataFrame:
    """
    Add columns: swing_high (bool), swing_low (bool).
    A swing high at bar i means high[i] > high[i-1] and high[i] > high[i+1].
    Same for swing low. Uses the simple 1-left 1-right rule from the Pine script.
    """
    h = df5["high"].values
    l = df5["low"].values
    n = len(df5)

    sh = np.zeros(n, dtype=bool)
    sl = np.zeros(n, dtype=bool)

    for i in range(1, n - 1):
        if h[i] > h[i - 1] and h[i] > h[i + 1]:
            sh[i] = True
        if l[i] < l[i - 1] and l[i] < l[i + 1]:
            sl[i] = True

    df5 = df5.copy()
    df5["swing_high"] = sh
    df5["swing_low"]  = sl
    return df5

def run_backtest(df1: pd.DataFrame,
                 df5: pd.DataFrame,
                 years: list,
                 session_start: dtime,
                 session_end: dtime,
                 rr: float,
                 max_stop) -> tuple[list, list]:
    """
    Returns (all_trades, trades_2026).
    Each trade is a dict:
      date, direction, entry, sl, tp, rr, result ('W'/'L')
    """

    # Pre-compute swings on all 5m data
    df5 = detect_5m_swings(df5)

    all_trades    = []
    trades_2026   = []

    # Iterate year by year
    for year in years:
        print(f"\n  ── Backtesting {year} ──")

        df1_y = df1[df1["datetime"].dt.year == year].copy()
        df5_y = df5[df5["datetime"].dt.year == year].copy()

        if df1_y.empty or df5_y.empty:
            print(f"    No data found for {year} — skipping.")
            continue

        trades_year = _backtest_year(df1_y, df5_y, session_start, session_end, rr, max_stop, year)
        all_trades.extend(trades_year)
        if year == 2026:
            trades_2026.extend(trades_year)

        # Year summary
        if trades_year:
            wins   = sum(1 for t in trades_year if t["result"] == "W")
            losses = len(trades_year) - wins
            wr     = wins / len(trades_year) * 100
            print(f"    Trades: {len(trades_year)}  |  W: {wins}  L: {losses}  |  Win-rate: {wr:.1f}%")
        else:
            print(f"    No trades taken in {year}.")

    return all_trades, trades_2026


def _backtest_year(df1: pd.DataFrame,
                   df5: pd.DataFrame,
                   session_start: dtime,
                   session_end: dtime,
                   rr: float,
                   max_stop,
                   year: int) -> list:

    trades = []

    # Group 1m data by trading date
    df1["date"] = df1["datetime"].dt.date
    trading_dates = sorted(df1["date"].unique())

    for date in trading_dates:
        day_trades = _backtest_day(
            df1[df1["date"] == date].reset_index(drop=True),
            df5[df5["datetime"].dt.date == date].reset_index(drop=True),
            session_start, session_end, rr, max_stop, date
        )
        trades.extend(day_trades)

    return trades


def _backtest_day(df1_day: pd.DataFrame,
                  df5_day: pd.DataFrame,
                  session_start: dtime,
                  session_end: dtime,
                  rr: float,
                  max_stop,
                  date) -> list:
    """
    Run the full strategy logic for a single trading day.
    State machine:
      IDLE → waiting for swing break
      SETUP_LONG / SETUP_SHORT → tracking CISD + FVG
      IN_TRADE → managing open position
    """
    trades = []

    # ── Build swing level list for the day from 5m ──
    # We'll replay them chronologically as 1m bars pass
    swing_highs = []  # list of (datetime, level)
    swing_lows  = []

    for _, row in df5_day.iterrows():
        if row["swing_high"]:
            swing_highs.append((row["datetime"], row["high"]))
        if row["swing_low"]:
            swing_lows.append((row["datetime"], row["low"]))

    # ── State ──
    state              = "IDLE"
    bias               = 0        # 1 = long, -1 = short
    setup_used         = False

    # Active swing levels (most recent unbroken)
    active_sh          = None     # (time, level)
    active_sl          = None

    # CISD state
    cisd_range_high    = None
    cisd_range_low     = None
    cisd_sl_level      = None
    cisd_start_idx     = None

    # Bear/bull run tracking
    bear_run_start     = None
    bear_run_count     = 0
    bear_run_open      = None
    bear_run_sl        = None

    bull_run_start     = None
    bull_run_count     = 0
    bull_run_open      = None
    bull_run_sl        = None

    # FVG
    fvg_valid          = False
    fvg_high           = None
    fvg_low            = None

    # Trade
    entry_price        = None
    sl_price           = None
    tp_price           = None
    direction          = None

    # Swing pointer for replaying as time passes
    sh_ptr = 0
    sl_ptr = 0

    def reset_cisd_fvg():
        nonlocal cisd_range_high, cisd_range_low, cisd_sl_level, cisd_start_idx
        nonlocal bear_run_start, bear_run_count, bear_run_open, bear_run_sl
        nonlocal bull_run_start, bull_run_count, bull_run_open, bull_run_sl
        nonlocal fvg_valid, fvg_high, fvg_low
        cisd_range_high = None;  cisd_range_low = None
        cisd_sl_level   = None;  cisd_start_idx = None
        bear_run_start  = None;  bear_run_count = 0
        bear_run_open   = None;  bear_run_sl    = None
        bull_run_start  = None;  bull_run_count = 0
        bull_run_open   = None;  bull_run_sl    = None
        fvg_valid = False;       fvg_high = None;  fvg_low = None

    n1 = len(df1_day)

    for i, row in df1_day.iterrows():
        bar_time = row["datetime"]
        t        = bar_time.time()
        o        = row["open"]
        h        = row["high"]
        l        = row["low"]
        c        = row["close"]

        in_entry_session = session_start <= t < session_end

        # ── Replay 5m swings that have formed by this 1m bar's close ──
        # A 5m swing at bar i is confirmed at the close of bar i+1
        while sh_ptr < len(swing_highs) and swing_highs[sh_ptr][0] <= bar_time:
            active_sh = swing_highs[sh_ptr]
            sh_ptr += 1
        while sl_ptr < len(swing_lows) and swing_lows[sl_ptr][0] <= bar_time:
            active_sl = swing_lows[sl_ptr]
            sl_ptr += 1

        # ── IN_TRADE: check TP / SL ──
        if state == "IN_TRADE":
            if direction == "long":
                if l <= sl_price:
                    result = "L"
                    trades.append(_make_trade(date, bar_time, direction, entry_price, sl_price, tp_price, rr, result))
                    state = "IDLE"; bias = 0; setup_used = False
                    reset_cisd_fvg()
                elif h >= tp_price:
                    result = "W"
                    trades.append(_make_trade(date, bar_time, direction, entry_price, sl_price, tp_price, rr, result))
                    state = "IDLE"; bias = 0; setup_used = False
                    reset_cisd_fvg()
            else:  # short
                if h >= sl_price:
                    result = "L"
                    trades.append(_make_trade(date, bar_time, direction, entry_price, sl_price, tp_price, rr, result))
                    state = "IDLE"; bias = 0; setup_used = False
                    reset_cisd_fvg()
                elif l <= tp_price:
                    result = "W"
                    trades.append(_make_trade(date, bar_time, direction, entry_price, sl_price, tp_price, rr, result))
                    state = "IDLE"; bias = 0; setup_used = False
                    reset_cisd_fvg()
            continue  # don't look for new setups while in a trade

        if setup_used:
            continue   # wait for a new swing break

        # ── SWING BREAK DETECTION ──
        if active_sh is not None and h > active_sh[1] and not setup_used:
            # Swing high broken → short bias
            bias = -1
            active_sh = None
            reset_cisd_fvg()
            state = "SETUP_SHORT"

        if active_sl is not None and l < active_sl[1] and not setup_used:
            # Swing low broken → long bias
            bias = 1
            active_sl = None
            reset_cisd_fvg()
            state = "SETUP_LONG"

        # ── CISD TRACKING ──
        if not in_entry_session:
            continue

        is_bearish = c < o
        is_bullish = c > o

        # ---- BULLISH CISD (long bias) ----
        if state == "SETUP_LONG":
            prev_is_bearish = df1_day.iloc[i - 1]["close"] < df1_day.iloc[i - 1]["open"] if i > 0 else False

            if is_bearish:
                if bear_run_start is None or not prev_is_bearish:
                    bear_run_start = i
                    bear_run_open  = o
                    bear_run_sl    = c
                    bear_run_count = 1
                else:
                    bear_run_count += 1
                    bear_run_sl     = min(bear_run_sl, c)
                    # Extend CISD if new lower low
                    if cisd_range_low is None or l < cisd_range_low:
                        cisd_start_idx  = bear_run_start
                        cisd_range_high = bear_run_open
                        cisd_range_low  = l
                        cisd_sl_level   = bear_run_sl
                        fvg_valid = False; fvg_high = None; fvg_low = None

                if bear_run_count >= 2 and cisd_start_idx is None:
                    cisd_start_idx  = bear_run_start
                    cisd_range_high = bear_run_open
                    cisd_range_low  = l
                    cisd_sl_level   = bear_run_sl

            else:
                bear_run_start = None
                bear_run_count = 0
                bear_run_open  = None
                bear_run_sl    = None

        # ---- BEARISH CISD (short bias) ----
        if state == "SETUP_SHORT":
            prev_is_bullish = df1_day.iloc[i - 1]["close"] > df1_day.iloc[i - 1]["open"] if i > 0 else False

            if is_bullish:
                if bull_run_start is None or not prev_is_bullish:
                    bull_run_start = i
                    bull_run_open  = o
                    bull_run_sl    = c
                    bull_run_count = 1
                else:
                    bull_run_count += 1
                    bull_run_sl     = max(bull_run_sl, c)
                    if cisd_range_high is None or h > cisd_range_high:
                        cisd_start_idx  = bull_run_start
                        cisd_range_low  = bull_run_open
                        cisd_range_high = h
                        cisd_sl_level   = bull_run_sl
                        fvg_valid = False; fvg_high = None; fvg_low = None

                if bull_run_count >= 2 and cisd_start_idx is None:
                    cisd_start_idx  = bull_run_start
                    cisd_range_low  = bull_run_open
                    cisd_range_high = h
                    cisd_sl_level   = bull_run_sl

            else:
                bull_run_start = None
                bull_run_count = 0
                bull_run_open  = None
                bull_run_sl    = None

        # ── FVG SCAN (3-bar window, candle[1] must be in CISD range) ──
        if cisd_start_idx is not None and not fvg_valid and i >= cisd_start_idx + 2:
            c1_idx = i - 2
            c2_idx = i - 1

            if cisd_start_idx <= c2_idx:
                c1 = df1_day.iloc[c1_idx]
                c2 = df1_day.iloc[c2_idx]
                c3 = row

                if state == "SETUP_LONG":
                    # Bearish FVG: c1 low = FVG high, c3 high = FVG low
                    fh = c1["low"]
                    fl = c3["high"]
                    if fh > fl:
                        fvg_high = fh; fvg_low = fl; fvg_valid = True

                elif state == "SETUP_SHORT":
                    # Bullish FVG: c1 high = FVG low, c3 low = FVG high
                    fl = c1["high"]
                    fh = c3["low"]
                    if fh > fl:
                        fvg_high = fh; fvg_low = fl; fvg_valid = True

        # ── ENTRY CHECK ──
        if cisd_range_high is not None and fvg_valid and in_entry_session:

            entered = False

            if state == "SETUP_LONG" and c > cisd_range_high:
                ep  = c
                slp = cisd_sl_level
                stop_size = ep - slp
                if stop_size <= 0:
                    pass
                elif max_stop is not None and stop_size > max_stop:
                    pass
                else:
                    tpp = ep + stop_size * rr
                    entry_price = ep; sl_price = slp; tp_price = tpp
                    direction = "long"
                    state = "IN_TRADE"; setup_used = True; bias = 0
                    entered = True

            elif state == "SETUP_SHORT" and c < cisd_range_low:
                ep  = c
                slp = cisd_sl_level
                stop_size = slp - ep
                if stop_size <= 0:
                    pass
                elif max_stop is not None and stop_size > max_stop:
                    pass
                else:
                    tpp = ep - stop_size * rr
                    entry_price = ep; sl_price = slp; tp_price = tpp
                    direction = "short"
                    state = "IN_TRADE"; setup_used = True; bias = 0
                    entered = True

    return trades


def _make_trade(date, entry_time, direction, entry, sl, tp, rr, result):
    return {
        "date":      str(date),
        "time":      entry_time.strftime("%H:%M"),
        "direction": direction,
        "entry":     round(entry, 4),
        "sl":        round(sl, 4),
        "tp":        round(tp, 4),
        "rr":        rr,
        "result":    result,
    }

def print_summary(all_trades: list, years: list):
    print("\n" + "═" * 50)
    print("  OVERALL SUMMARY")
    print("═" * 50)
    if not all_trades:
        print("  No trades taken across all years.")
        return

    df = pd.DataFrame(all_trades)
    df["year"] = pd.to_datetime(df["date"]).dt.year

    for yr in sorted(df["year"].unique()):
        sub = df[df["year"] == yr]
        w   = (sub["result"] == "W").sum()
        l   = (sub["result"] == "L").sum()
        wr  = w / len(sub) * 100 if len(sub) > 0 else 0
        net_r = w * sub.iloc[0]["rr"] - l  # net R gained
        print(f"\n  {yr}")
        print(f"    Total trades : {len(sub)}")
        print(f"    Wins         : {w}")
        print(f"    Losses       : {l}")
        print(f"    Win rate     : {wr:.1f}%")
        print(f"    Net R        : {net_r:+.2f}R")

    w_all  = (df["result"] == "W").sum()
    l_all  = (df["result"] == "L").sum()
    wr_all = w_all / len(df) * 100
    rr_val = df.iloc[0]["rr"]
    net_r_all = w_all * rr_val - l_all
    print(f"\n  ALL YEARS COMBINED")
    print(f"    Total trades : {len(df)}")
    print(f"    Wins         : {w_all}")
    print(f"    Losses       : {l_all}")
    print(f"    Win rate     : {wr_all:.1f}%")
    print(f"    Net R        : {net_r_all:+.2f}R")
    print("═" * 50)

def prompt(label: str, default=None) -> str:
    if default is not None:
        val = input(f"  {label} [{default}]: ").strip()
        return val if val else str(default)
    return input(f"  {label}: ").strip()


def get_inputs():
    print()
    print("╔══════════════════════════════════════════════════════╗")
    print("║    5m Break → 1m CISD + FVG  |  Backtester          ║")
    print("╚══════════════════════════════════════════════════════╝")
    print()

    # CSV paths — ask for filename only, build full path from DATA_DIR
    print(f"  Data folder: {DATA_DIR}")
    fname_1m = prompt("1m CSV filename (e.g. NQ_1m.csv)")
    fname_5m = prompt("5m CSV filename (e.g. NQ_5m.csv)")
    path_1m  = os.path.join(DATA_DIR, fname_1m)
    path_5m  = os.path.join(DATA_DIR, fname_5m)

    # Years
    years_raw = prompt("Years to backtest (comma-separated, e.g. 2023,2024,2025)", "2024,2025,2026")
    years = [int(y.strip()) for y in years_raw.split(",") if y.strip().isdigit()]
    if not years:
        raise ValueError("No valid years provided.")

    # Session
    session_raw = prompt("Entry session window (HH:MM-HH:MM)", DEFAULT_SESSION)
    session_start, session_end = parse_session(session_raw if session_raw else DEFAULT_SESSION)

    # RR
    rr_raw = prompt("RR target (e.g. 1 or 1.5)", str(DEFAULT_RR))
    rr     = float(rr_raw) if rr_raw else DEFAULT_RR

    # Max stop
    ms_raw = prompt("Max stop in points (blank = no limit)", "")
    max_stop = float(ms_raw) if ms_raw.strip() else None

    print()
    print(f"  1m data    : {path_1m}")
    print(f"  5m data    : {path_5m}")
    print(f"  Years      : {years}")
    print(f"  Session    : {session_start.strftime('%H:%M')} – {session_end.strftime('%H:%M')} ET")
    print(f"  RR         : {rr}")
    print(f"  Max stop   : {'unlimited' if max_stop is None else max_stop}")
    print()

    return path_1m, path_5m, years, session_start, session_end, rr, max_stop

def main():
    try:
        (path_1m, path_5m,
         years, session_start, session_end,
         rr, max_stop) = get_inputs()
    except (ValueError, KeyboardInterrupt) as e:
        print(f"\n  Error: {e}")
        sys.exit(1)

    print("Loading data...")
    try:
        df1 = load_csv(path_1m, "1m")
        df5 = load_csv(path_5m, "5m")
    except Exception as e:
        print(f"\n  Failed to load data: {e}")
        sys.exit(1)

    print("\nRunning backtest...")
    all_trades, trades_2026 = run_backtest(
        df1, df5, years,
        session_start, session_end,
        rr, max_stop
    )

    print_summary(all_trades, years)

    if 2026 in years:
        # Save to same folder as the 1m CSV
        out_dir = str(Path(path_1m).parent)
        save_2026(trades_2026, out_dir)

    print("\nDone.\n")


if __name__ == "__main__":
